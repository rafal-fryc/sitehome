  Simple timeline implementation:
  ```tsx
  // Group cases by year
  const casesByYear = useMemo(() => {
    const map = new Map<number, number>();
    for (const c of pattern.cases) {
      map.set(c.year, (map.get(c.year) || 0) + 1);
    }
    return [...map.entries()].sort((a, b) => a[0] - b[0]);
  }, [pattern.cases]);
  const maxCount = Math.max(...casesByYear.map(([, c]) => c));

  // Render as horizontal bars
  {casesByYear.map(([year, count]) => (
    <div key={year} className="flex items-center gap-2 text-sm">
      <span className="w-12 text-right text-muted-foreground">{year}</span>
      <div className="h-4 bg-gold/60 rounded" style={{ width: `${(count / maxCount) * 100}%`, minWidth: '8px' }} />
      <span className="text-muted-foreground">{count}</span>
    </div>
  ))}
  ```
