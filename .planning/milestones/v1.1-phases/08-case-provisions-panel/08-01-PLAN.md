```typescript
import { useQuery } from "@tanstack/react-query";

export function useCaseFile(caseId: string | null) {
  return useQuery({
    queryKey: ["case-file", caseId],
    queryFn: async () => {
      const res = await fetch(`/data/ftc-files/${caseId}.json`);
      if (!res.ok) throw new Error("Failed to load case file");
      return res.json();
    },
    enabled: !!caseId,
    staleTime: Infinity,
  });
}
```

No custom return type needed -- the case file shape is used dynamically in the Sheet component. Keep it simple like the existing hooks.

**Implementation details:**

1. **Width override:** `SheetContent className="w-full sm:w-[50vw] sm:max-w-2xl"` -- overrides default `w-3/4 sm:max-w-sm` which is too narrow for provision text.

2. **Data fetching:** Call `useCaseFile(open ? caseData?.id ?? null : null)` -- only fetch when Sheet is open. This prevents stale queries.

3. **Header:** Inside SheetHeader:
   - SheetTitle: `caseData.company_name` with `font-garamond`
   - SheetDescription: Compose from parts:
     - `{caseData.year}` | `{caseData.violation_type}` (uppercase first letter) | `{caseData.num_provisions} provisions`
     - FTC.gov link: `<a href={caseData.ftc_url} target="_blank" rel="noopener noreferrer" className="text-gold hover:text-gold/80 inline-flex items-center gap-1">View on FTC.gov <ExternalLink className="h-3 w-3" /></a>`

4. **Grouping logic:** Create a `groupByTopic` function (can be a local helper or inline useMemo):
   ```typescript
   function groupByTopic(provisions: any[]): Map<string, any[]> {
     const groups = new Map<string, any[]>();
     for (const prov of provisions) {
       const topics = prov.statutory_topics ?? [];
       if (topics.length === 0) {
         const key = "Other";
         if (!groups.has(key)) groups.set(key, []);
         groups.get(key)!.push(prov);
       } else {
         for (const topic of topics) {
           if (!groups.has(topic)) groups.set(topic, []);
           groups.get(topic)!.push(prov);
         }
       }
     }
     return groups;
   }
   ```
   Wrap in useMemo keyed on the fetched data's provisions array.

5. **Provision sort within groups:** Preserve original order (order in JSON array = order in consent decree). Do NOT re-sort within groups.

6. **Grouped list rendering:**
   ```
   For each [topic, provisions] in groups:
     <div key={topic}>
       <div className="flex items-center gap-2 px-3 py-2 bg-primary/5 border-b border-rule">
         <h3 className="text-sm font-semibold font-garamond text-primary uppercase tracking-wide-label">{topic}</h3>
         <Badge variant="secondary" className="text-xs">{provisions.length}</Badge>
       </div>
       {provisions.map(prov => <ProvisionRow key={`${topic}-${prov.provision_number}`} provision={prov} />)}
     </div>
   ```

7. **Loading state:** When `isLoading`, show 3-4 skeleton rows inside the body area:

   ```
   <div className="space-y-3 animate-pulse">

       </div>
     ))}
   </div>
   ```
